2.3 Scroll Animation using IntersectionObserver

``` html
<!DOCTYPE html>
<html>
<head>
    <style>
        body { margin: 0; font-family: Arial; }
        .spacer { height: 100vh; background: #f0f0f0; }
        .target {
            width: 300px;
            height: 300px;
            background: #667eea;
            margin: 50px auto;
            display: flex;
            align-items: center;
            justify-content: center;
            color: white;
            font-size: 18px;
            opacity: 0.3;
            transform: scale(0.8);
            transition: all 0.6s ease;
        }
        .target.visible {
            opacity: 1;
            transform: scale(1);
        }
    </style>
</head>
<body>
    <div class="spacer">Role para baixo</div>
    <div class="target">Elemento 1</div>
    <div class="spacer"></div>
    <div class="target">Elemento 2</div>
    <div class="spacer"></div>
    <div class="target">Elemento 3</div>
    <div class="spacer"></div>

    <script>
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                } else {
                    entry.target.classList.remove('visible');
                }
            });
        }, {
            threshold: 0.5 // 50% do elemento deve estar visível
        });

        // Observar todos os elementos .target
        document.querySelectorAll('.target').forEach(target => {
            observer.observe(target);
        });
    </script>
</body>
</html>
```

2.3 Scroll Animation para uma barra de progresso

```html
<!DOCTYPE html>
<html>
<head>
    <style>
        body { 
            margin: 0; 
            font-family: Arial;
            height: 300vh; /* conteúdo longo para scroll */
        }
        
        .progress-bar {
            position: fixed;
            top: 0;
            left: 0;
            height: 4px;
            background: linear-gradient(90deg, #667eea, #764ba2);
            width: 0%;
            transition: width 0.1s ease;
            z-index: 1000;
        }
        
        .content {
            padding: 20px;
            max-width: 800px;
            margin: 0 auto;
        }
        
        .section {
            height: 100vh;
            display: flex;
            align-items: center;
            justify-content: center;
            font-size: 2em;
            border-bottom: 1px solid #eee;
        }
    </style>
</head>
<body>
    <div class="progress-bar"></div>
    
    <div class="content">
        <div class="section">Secção 1</div>
        <div class="section">Secção 2</div>
        <div class="section">Secção 3</div>
    </div>

    <script>
        const progressBar = document.querySelector('.progress-bar');
        
        // Throttle para performance
        let ticking = false;
        
        function updateProgressBar() {
            const scrollTop = window.pageYOffset;
            const docHeight = document.documentElement.scrollHeight - window.innerHeight;
            const scrollPercent = (scrollTop / docHeight) * 100;
            
            progressBar.style.width = scrollPercent + '%';
            ticking = false;
        }
        
        window.addEventListener('scroll', () => {
            if (!ticking) {
                requestAnimationFrame(updateProgressBar);
                ticking = true;
            }
        });
    </script>
</body>
</html>
```

2.3 Exemplo de animações simultâneas com delay sequenciais desencadeados pela scroll. Exemplo com 4 cartões


> [!NOTE] Title
> Contents

```html
<!DOCTYPE html>
<html>
<head>
    <style>
        body { margin: 0; padding: 20px; font-family: Arial; }
        
        .container {
            max-width: 1200px;
            margin: 0 auto;
            padding: 100px 0;
        }
        
        .card {
            width: 300px;
            height: 200px;
            background: white;
            border-radius: 10px;
            box-shadow: 0 10px 30px rgba(0,0,0,0.1);
            margin: 50px;
            display: inline-block;
            opacity: 0;
            transform: translateY(0px);
            transition: all 0.8s cubic-bezier(0.25, 0.46, 0.45, 0.94);
        }
      
        .card.animate {
            opacity: 1;
            transform: translateY(60px) rotate(10deg);
        }
        
        /* Delays escalonados */
        .card:nth-child(1).animate { transition-delay: 0.1s; }
        .card:nth-child(2).animate { transition-delay: 0.5s; }
        .card:nth-child(3).animate { transition-delay: 1s; }
        .card:nth-child(4).animate { transition-delay: 1.4s; }
        
        .spacer { height: 100vh; }
    </style>
</head>
<body>
    <div class="spacer">Role para baixo</div>
    
    <div class="container">
        <div class="card">Card 1</div>
        <div class="card">Card 2</div>
        <div class="card">Card 3</div>
        <div class="card">Card 4</div>
    </div>
    
    <div class="spacer"></div>

    <script>
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    // Animar todos os cards do container
                    const cards = entry.target.querySelectorAll('.card');
                    cards.forEach(card => card.classList.add('animate'));
                }
            });
        }, { threshold: 0.2 });

        observer.observe(document.querySelector('.container'));
    </script>
</body>
</html>
```

2.3 Exemplo de rotação com base em scroll

```html
<!DOCTYPE html>
<html>
<head>
    <style>
        body { 
            margin: 0; 
            font-family: Arial;
            height: 400vh;
            background: linear-gradient(to bottom, #667eea, #764ba2);
        }
        
        .rotating-element {
            position: fixed;
            top: 50%;
            left: 50%;
            width: 200px;
            height: 200px;
            background: white;
            border-radius: 20px;
            box-shadow: 0 20px 60px rgba(0,0,0,0.3);
            display: flex;
            align-items: center;
            justify-content: center;
            font-size: 18px;
            font-weight: bold;
            transform: translate(-50%, -50%);
            will-change: transform;
        }
        
        .scroll-indicator {
            position: fixed;
            bottom: 50px;
            left: 50%;
            transform: translateX(-50%);
            color: white;
            font-size: 18px;
            text-align: center;
        }
    </style>
</head>
<body>
    <div class="rotating-element">
        Role a página
    </div>
    
    <div class="scroll-indicator">
        <div>Progresso: <span id="progress">0</span>%</div>
        <div>Rotação: <span id="rotation">0</span>°</div>
    </div>

    <script>
        const rotatingElement = document.querySelector('.rotating-element');
        const progressSpan = document.getElementById('progress');
        const rotationSpan = document.getElementById('rotation');
        
        function updateRotation() {
            const scrollTop = window.pageYOffset;
            const docHeight = document.documentElement.scrollHeight - window.innerHeight;
            const scrollFraction = scrollTop / docHeight;
            
            // Rotação completa (360°) ao chegar ao fim
            const rotation = scrollFraction * 360;
            
            // Escala baseada no scroll (0.5x a 1.5x)
            const scale = 0.5 + (scrollFraction * 1);
            
            rotatingElement.style.transform = 
                `translate(-50%, -50%) rotate(${rotation}deg) scale(${scale})`;
            
            // Atualizar indicadores
            progressSpan.textContent = Math.round(scrollFraction * 100);
            rotationSpan.textContent = Math.round(rotation);
            ticking = false;
        }
        
        let ticking = false;
        window.addEventListener('scroll', () => {
            if (!ticking) {
                requestAnimationFrame(updateRotation);
                ticking = true;
            }
        });
    </script>
</body>
</html>
```


2.6 Parallax

```html
<!DOCTYPE html>
<html>
<head>
    <style>
        body { margin: 0; font-family: Arial; }
        
        
        /* Método alternativo com JavaScript */
        .hero {
            height: 100vh;
            background: url('data:image/svg+xml,<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100"><circle cx="20" cy="20" r="2" fill="%23fff" opacity="0.3"/><circle cx="80" cy="40" r="1" fill="%23fff" opacity="0.5"/><circle cx="40" cy="80" r="1.5" fill="%23fff" opacity="0.4"/></svg>') #667eea;
            display: flex;
            align-items: center;
            justify-content: center;
            color: white;
            font-size: 3em;
            position: relative;

            padding: 50px;
            position: relative;
            z-index: 1;
        }
    </style>

<body>
    <div class="hero">
        <h1>Scroll Parallax</h1>
    </div>
    
    <div class="content">
        <h2>Conteúdo Principal</h2>
        <p>Lorem ipsum dolor sit amet, consectetur adipiscing elit...</p>

        const hero = document.querySelector('.hero');
        
        function updateParallax() {
            const scrolled = window.pageYOffset;
            const parallax = scrolled * 0.5;
            
            hero.style.transform = `translateY(${parallax}px)`;
            ticking = false;
        }
        
        let ticking = false;
        window.addEventListener('scroll', () => {
            if (!ticking) {
                requestAnimationFrame(updateParallax);
                ticking = true;
            }
        });
    </script>
</body>
</html>
```
